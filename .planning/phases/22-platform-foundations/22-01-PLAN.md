Medium-risk sites use hardcoded constants but benefit from defense-in-depth. Internal/CDP/localhost calls must NOT be converted — the guard blocks private IPs by default.

```typescript
// src/infra/net/fetch-guard.ts — full public API
export type GuardedFetchOptions = {
  url: string;
  fetchImpl?: FetchLike; // Optional override for testing
  init?: RequestInit; // Standard fetch init options
  maxRedirects?: number; // Default: 3
  timeoutMs?: number; // Optional abort timeout
  signal?: AbortSignal; // Optional external abort signal
  policy?: SsrFPolicy; // Optional hostname allowlist
  lookupFn?: LookupFn; // For testing only
  pinDns?: boolean; // Default: true
  auditContext?: string; // Label for SSRF block logs
};

export type GuardedFetchResult = {
  response: Response;
  finalUrl: string; // URL after redirects
  release: () => Promise<void>; // MUST call in finally block
};

export async function fetchWithSsrFGuard(params: GuardedFetchOptions): Promise<GuardedFetchResult>;
```
