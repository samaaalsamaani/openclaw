    - Name: `graph_trace`
    - Description: "Traverse causal graph edges (CAUSED_BY, LED_TO, SUPPORTS) to trace root causes and consequences of a topic"
    - Input schema:
      ```json
      {
        "type": "object",
        "properties": {
          "entity": { "type": "string", "description": "Entity name or topic to trace" },
          "direction": { "type": "string", "enum": ["causes", "effects", "both"], "description": "Traversal direction: causes=what caused this, effects=what followed from this, both=full chain" },
          "hops": { "type": "number", "description": "Max traversal depth 1-5 (default 2)" }
        },
        "required": ["entity"]
      }
      ```
    - Handler logic:
      ```javascript
      const neo4j = require("neo4j-driver");
      const BOLT_URI = process.env.PAIOS_GRAPH_URI || "bolt://localhost:7687";
      const direction = args.direction || "both";
      const hops = Math.max(1, Math.min(5, Number(args.hops) || 2));
      const entity = args.entity;

      const driver = neo4j.driver(BOLT_URI, neo4j.auth.none());
      const session = driver.session();
      try {
        let query;
        if (direction === "causes") {
          query = `MATCH (n {name: $entity})<-[:CAUSED_BY|LED_TO|SUPPORTS*1..${hops}]-(cause)
                   RETURN cause.name AS name, cause.type AS type, cause.description AS description
                   LIMIT 10`;
        } else if (direction === "effects") {
          query = `MATCH (n {name: $entity})-[:CAUSED_BY|LED_TO|SUPPORTS*1..${hops}]->(effect)
                   RETURN effect.name AS name, effect.type AS type, effect.description AS description
                   LIMIT 10`;
        } else {
          query = `MATCH path = (n {name: $entity})-[:CAUSED_BY|LED_TO|SUPPORTS*1..${hops}]-(related)
                   RETURN related.name AS name, related.type AS type, related.description AS description,
                          [rel in relationships(path) | type(rel)] AS edge_types
                   LIMIT 15`;
        }
        const result = await session.run(query, { entity });
        const records = result.records.map(r => r.toObject());
        return { content: [{ type: "text", text: JSON.stringify(records, null, 2) }] };
      } finally {
        await session.close();
        await driver.close();
      }
      ```

    The handler must be wrapped in a try/catch so errors return a structured error response (consistent with other KB MCP tools).
